    基本使用

    通过 content 设置要复制的文字内容，通过 alertStyle 设置提示样式

    
  <template>
      <view>
          <up-copy content="uview-plus is great !">
              <text>点击复制</text>
          </up-copy>
          <up-copy content="uview-plus is great !">
              <up-button type="primary">点击复制</up-button>
          </up-copy>
      </view>
  </template>
